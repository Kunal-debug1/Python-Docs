03 · Expressions

Operators

Python provides rich operator support across arithmetic, comparison, logical, bitwise, and identity domains.

PYTHON
# Arithmetic operators
print(10 + 3)   # 13  → addition
print(10 - 3)   # 7   → subtraction
print(10 * 3)   # 30  → multiplication
print(10 / 3)   # 3.333 → true division (always float)
print(10 // 3)  # 3   → floor division
print(10 % 3)   # 1   → modulus (remainder)
print(2 ** 8)   # 256 → exponentiation

# Comparison operators
print(5 == 5)   # True
print(5 != 4)   # True
print(5 >= 5)   # True

# Logical operators
print(True and False)  # False
print(True or  False)  # True
print(not True)        # False

# Identity and membership
x = [1,2,3]

print(0b1010 | 0b1100) # 14 → OR
print(1 << 3)           # 8  → left shift

# Walrus operator := (Python 3.8+)

04 · Control Flow

If / Elif / Else

Conditional statements let your program make decisions. Python uses indentation (4 spaces) — not braces — to define code blocks.

PYTHON
# Basic if / elif / else
score = 78
if score >= 90:
    grade = "A"
elif score >= 80:
    grade = "B"
elif score >= 70:
    grade = "C"
else:
    grade = "D"

# Ternary (conditional expression)
label = "Pass" if score >= 50 else "Fail"

# Chained comparisons (Python exclusive)
print(0 < score < 100)  # True

# match / case (Python 3.10+ — structural pattern matching)
command = "quit"
match command:
    case "quit":
        print("Exiting...")
    case "hello" | "hi":
        print("Hello!")
    case _:                   # wildcard (default)
        print("Unknown command")
📌
Truthy / Falsy: 0, "", [], {}, None, False are all falsy. Everything else is truthy. Use if x: instead of if x != None:.
05 · Iteration

Loops

Python has for (iterate over any iterable) and while (repeat while condition holds). Both support break, continue, and else clauses.

PYTHON
# for loop with range
for i in range(1, 6):     # 1 2 3 4 5
    print(i, end=" ")

# iterate with index — use enumerate()
fruits = ["apple", "banana", "cherry"]
for idx, fruit in enumerate(fruits):
    print(f"{idx}: {fruit}")

# iterate two lists together — zip()
names = ["A", "B"]; scores = [90, 85]
for name, score in zip(names, scores):
    print(f"{name}: {score}")

# while loop
count = 0
while count < 5:
    count += 1

# break and continue
for n in range(10):
    if n == 3: continue     # skip 3
    if n == 7: break        # stop at 7
    print(n, end=" ")       # 0 1 2 4 5 6

# for...else (else runs when loop completes without break)
for i in range(5):
    if i == 10: break
else:
    print("No break occurred")  # prints

# reversed() and sorted()
for x in reversed([1,2,3]):
    print(x)                   # 3 2 1

Comprehensions

Compact, readable syntax for constructing lists, dicts, sets, and generator expressions in a single line.

10 · Functions

Functions

Functions are first-class objects in Python — they can be stored in variables, passed as arguments, and returned from other functions.

PYTHON
# Basic function with defaults and type hints
def greet(name: str, greeting: str = "Hello") -> str:
    """Return a formatted greeting string."""
    return f"{greeting}, {name}!"

# *args — variable positional arguments
def add(*numbers):
    return sum(numbers)
print(add(1, 2, 3, 4))  # 10

# **kwargs — variable keyword arguments
def show_info(**details):
    for k, v in details.items():
        print(f"{k}: {v}")

# Keyword-only arguments (after *)
def create_tag(content, *, tag="div", cls=""):
    return f'<{tag} class="{cls}">{content}</{tag}>'

def min_max(lst):
    return min(lst), max(lst)
lo, hi = min_max([3,1,4,1,5])

# Closures
def make_multiplier(factor):
    def multiply(x):
        return x * factor   # closes over factor
    return multiply
double = make_multiplier(2)
print(double(5))  # 10

# functools — useful function tools
from functools import lru_cache, partial

@lru_cache(maxsize=None)   # memoize results
def fib(n):
    if n < 2: return n
    return fib(n-1) + fib(n-2)

double = partial(add, 2)    # fix first argument
11 · Object Oriented

Classes & OOP

Python is fully object-oriented. Classes bundle data (attributes) and behaviour (methods) together using __init__ and self.

PYTHON
class BankAccount:
    """A simple bank account."""
    interest_rate = 0.05       # class variable (shared)

    def __init__(self, owner: str, balance: float = 0):
        self.owner   = owner   # instance variables
        self._balance = balance  # _ = "private by convention"

    @property
    def balance(self):          # getter
        return self._balance

    def deposit(self, amount: float):
        if amount <= 0: raise ValueError("Amount must be positive")
        self._balance += amount

    def withdraw(self, amount: float):
        if amount > self._balance: raise ValueError("Insufficient funds")
        self._balance -= amount

    @classmethod
    def set_rate(cls, rate):    # operates on class, not instance
        cls.interest_rate = rate

    @staticmethod
    def validate(amount):      # no self or cls
        return amount > 0

    def __repr__(self):         # machine-readable representation
        return f"BankAccount({self.owner!r}, {self._balance})"

    def __str__(self):          # human-readable (for print)
        return f"{self.owner}'s account: ₹{self._balance:,.2f}"

acc = BankAccount("Kunal", 5000)
acc.deposit(1000)
print(acc)           # Kunal's account: ₹6,000.00
💡
Dataclasses (Python 3.7+) auto-generate __init__, __repr__, and __eq__ for you. Use @dataclass decorator for simple data-holding classes.
PYTHON
from dataclasses import dataclass, field

@dataclass
class Student:
    name:   str
    age:    int
    grades: list = field(default_factory=list)

    def average(self) -> float:
        return sum(self.grades) / len(self.grades) if self.grades else 0

s = Student("Priya", 21, [90, 85, 92])
print(s.average())  # 89.0
12 · OOP Continued

Inheritance

Inheritance lets a child class reuse and extend a parent class. Python supports multiple inheritance and provides super() for cooperative method resolution.

PYTHON
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        raise NotImplementedError   # abstract method

    def __str__(self):
        return f"{self.__class__.__name__}({self.name})"

class Dog(Animal):
    def speak(self):
        return "Woof!"

class Cat(Animal):
    def speak(self):
        return "Meow!"

# Polymorphism
animals = [Dog("Rex"), Cat("Luna"), Dog("Buddy")]
for a in animals:
    print(f"{a}: {a.speak()}")

# Abstract base classes
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self) -> float: ...
    @abstractmethod
    def perimeter(self) -> float: ...

class Circle(Shape):
    def __init__(self, r): self.r = r
    def area(self):       return 3.14159 * self.r ** 2
    def perimeter(self): return 2 * 3.14159 * self.r

# issubclass / isinstance
print(issubclass(Dog, Animal))   # True
print(isinstance(Dog("x"), Animal)) # True
13 · Advanced Functions

Decorators Advanced

A decorator is a function that wraps another function to extend its behaviour without modifying its source code.

PYTHON
import functools, time

# Basic decorator
def timer(func):
    @functools.wraps(func)       # preserve metadata
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        end   = time.perf_counter()
        print(f"{func.__name__} took {end - start:.4f}s")
        return result
    return wrapper

@timer
def slow_function():
    time.sleep(0.1)

# Decorator with arguments (factory pattern)
def repeat(n):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for _ in range(n):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(3)
def say_hi(): print("Hi!")

# Stacking decorators (applied bottom-up)
@timer
@repeat(2)
def double_hello(): print("Hello")

# Class decorators
def singleton(cls):
    instances = {}
    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return get_instance

16 · Robustness

Error Handling

Python uses exceptions for error signalling. Always handle exceptions at the appropriate level — not too broad, not too narrow.

PYTHON
# try / except / else / finally
try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"Error: {e}")
except (TypeError, ValueError) as e:
    print(f"Type or value error: {e}")
else:
    print("Success!")        # runs if no exception
finally:
    print("Always runs")    # cleanup

# Custom exceptions
class InsufficientFundsError(Exception):
    def __init__(self, amount, available):
        self.amount    = amount
        self.available = available
        super().__init__(
            f"Need ₹{amount}, only ₹{available} available")

# raise and re-raise
def withdraw(balance, amount):
    if amount > balance:
        raise InsufficientFundsError(amount, balance)

try:
    withdraw(100, 200)
except InsufficientFundsError as e:
    print(e)
    raise                   # re-raise for caller

# Exception hierarchy
# BaseException → Exception → ValueError, TypeError,
#   KeyError, IndexError, FileNotFoundError, etc.

# Context managers (__enter__ / __exit__)
from contextlib import contextmanager

@contextmanager
def managed_resource(name):
    print(f"Acquiring {name}")
    try:
        yield name
    finally:
        print(f"Releasing {name}")

with managed_resource("DB connection") as res:
    print(f"Using {res}")
17 · Advanced

Iterators & Generators

Iterators implement the __iter__ / __next__ protocol. Generators use yield to produce values lazily — ideal for large data.

PYTHON
# Generator function
def countdown(n):
    while n > 0:
        yield n           # suspends here, resumes on next()
        n -= 1

for i in countdown(3):
    print(i)            # 3 2 1

# Infinite generator
def fibonacci():
    a, b = 0, 1
    while True:
        yield a
        a, b = b, a + b

import itertools
first10 = list(itertools.islice(fibonacci(), 10))

# Custom iterator class
class Range:
    def __init__(self, start, stop):
        self.current = start; self.stop = stop
    def __iter__(self): return self
    def __next__(self):
        if self.current >= self.stop: raise StopIteration
        val = self.current; self.current += 1; return val

# itertools — powerful iteration tools
import itertools as it
list(it.chain([1,2], [3,4]))     # [1,2,3,4]
list(it.product("AB", repeat=2))  # all combos
list(it.combinations([1,2,3],2)) # [(1,2),(1,3),(2,3)]
list(it.permutations([1,2,3],2)) # 6 pairs
list(it.groupby(["a","a","b"], key=lambda x:x))

Lambda & Functional Tools

Python supports functional patterns through lambda expressions, map(), filter(), reduce(), and the functools module.

19 · Concurrency

Async / Await Advanced

Python's asyncio provides cooperative multitasking for I/O-bound workloads — HTTP calls, database queries, file reads — without threads.

PYTHON
import asyncio

# async def defines a coroutine
async def fetch_data(url):
    print(f"Fetching {url}...")
    await asyncio.sleep(1)   # simulate I/O wait
    return {"url": url, "data": "..."}

# Run multiple coroutines concurrently
async def main():
    urls = ["api.example.com/1", "api.example.com/2"]
    
    # Sequential (slow)
    for url in urls:
        result = await fetch_data(url)

    # Concurrent (fast) — all run at once
    tasks   = [fetch_data(u) for u in urls]
    results = await asyncio.gather(*tasks)

    # gather with exception handling
    results = await asyncio.gather(*tasks, return_exceptions=True)

asyncio.run(main())   # entry point

# Async context manager
async def read_file(path):
    async with open(path) as f:   # requires aiofiles
        return await f.read()

# Async generator
async def ticker(delay, count):
    for i in range(count):
        yield i
        await asyncio.sleep(delay)

async def consume():
    async for val in ticker(0.5, 5):
        print(val)
20 · Text Processing

Regular Expressions

The re module provides Perl-compatible regex for pattern matching, searching, splitting, and replacing text.

PYTHON
import re

text = "Contact: kunal@example.com or admin@test.org"

# re.search() — find first match anywhere
m = re.search(r'\b\w+@\w+\.\w+\b', text)
print(m.group())          # kunal@example.com
print(m.span())           # (10, 27)

# re.findall() — all non-overlapping matches
emails = re.findall(r'[\w.]+@[\w.]+\.\w+', text)

# re.sub() — replace matches
masked = re.sub(r'[\w.]+@[\w.]+\.\w+', "[REDACTED]", text)

# Compile for reuse (faster in loops)
pattern = re.compile(r'(\d{4})-(\d{2})-(\d{2})')
m = pattern.match("2025-05-21")
year, month, day = m.groups()

# Common patterns
# \d   digit        \D non-digit
# \w   word char    \W non-word
# \s   whitespace   \S non-whitespace
# .    any char     ^ start  $ end
# *    0 or more    + 1 or more   ? optional
# {n}  exactly n    {n,m} between n and m
# []   char class   |  alternation
# ()   group        (?:) non-capturing group
# (?P<name>) named group

re.findall(r'python', text, re.IGNORECASE)

22 · Networking

API Requests

Use the requests library for synchronous HTTP. For async, use httpx or aiohttp.

PYTHON
import requests

# GET request
response = requests.get(
    "https://api.example.com/users",
    params={"page": 1, "limit": 10},
    headers={"Authorization": "Bearer TOKEN"},
    timeout=10
)
response.raise_for_status()   # throws on 4xx/5xx
data = response.json()

# POST — send JSON body
res = requests.post(
    "https://api.example.com/users",
    json={"name": "Kunal", "email": "k@example.com"},
    headers={"Authorization": "Bearer TOKEN"}
)

# Session (reuse connections + headers)
with requests.Session() as s:
    s.headers.update({"Authorization": "Bearer TOKEN"})
    r1 = s.get("/users")
    r2 = s.get("/posts")

# Full error handling
try:
    r = requests.get("https://api.example.com/data", timeout=5)
    r.raise_for_status()
    return r.json()
except requests.exceptions.HTTPError as e:
    print(f"HTTP {r.status_code}: {e}")
except requests.exceptions.Timeout:
    print("Request timed out")
except requests.exceptions.ConnectionError:
    print("Network unreachable")

# Async HTTP with httpx
import httpx, asyncio

async def fetch(url):
    async with httpx.AsyncClient() as client:
        r = await client.get(url)
        return r.json()

Python ships with "batteries included" — hundreds of modules covering dates, math, OS, threading, and more.

datetime · timedelta · timezone

collections
